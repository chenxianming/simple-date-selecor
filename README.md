# simple-date-selecor
Date selector for QuickCMS

    #
    @Example 

    <div class="simpledateselector">
        <input type="text" />
    </div>
    
    

    var els = document.getElementsByClassName('simpledateselector')[0],
        selector = new simpleDateSelector(els);
        selector.init('pass time to input');
